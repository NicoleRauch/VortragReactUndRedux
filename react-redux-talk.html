const USER_ADDED = "USER_ADDED";                  
                  
function addUser(user) {
  return {
    type: USER_ADDED,
    payload: user
  }
}

    Reducers

    
function users(state = INITIAL_STATE.users, action = {}) {
  switch (action.type) {
    case Events.USER_ADDED:
      return state.concat(action.payload);
  }
  return state;
}

export default combineReducers({
  users
});
